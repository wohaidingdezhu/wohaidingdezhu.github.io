# this在类中的概念

/**
 * 类的本质就是函数 就是一个壳子
 */
// 类 class xxxx -> 容器/作用域/模块 -> 壳子
class Test1 {
  constructor() {}
  say() {}
  static do() {}
}
const test1 = new Test1();

// 函数
function Test2() {
  // new -> constructor
}
Test2.prototype.say = function () {};
Test2.do = function () {};
const Test2 = new Test2();

const Test = (function () {
  function Test2() {
    // new -> constructor
  }
  Test.prototype.say = function () {};
  Test.do = function () {};
  window.Test = Test;
})();

class Test {
  //构造器
  constructor() {
    // 类的非静态方法 new -> this -> {}
    this.test = function () {
      console.log("none-static:" + this);
    };
  }
  // 没执行 静态方法 类原型上面的静态属性 不是在new过程中做
  // 定义的时候放到Test.prototype{...}
  // new 的时候 生成this的新的指向 {} -> __proto__ -> Test.prototype -> __proto__ -> Object.prototype -> null
  // Test -> prototype {} -> test 方法
  test() {
    console.log("static", this); //Test
  }
}
// this -> {}
const test = new Test();
// console.log(test);
test.test();

const TestA = Object.create(null);
console.log(TestA); //no properties

const TestB = {};
console.log(TestB); //Prototype

class Father {
  constructor(age) {
    // new -> this -> { } -> age属性
    // no this binding 没有实例化 绑定不上
    this.age = 16;
    // this.age = age;
  }
  swim() {
    console.log("go swimming!!!");
  }
}
class Son extends Father {
  constructor( age ) {
    // 调用father上面的constructor
    // 生成this绑定,Father this -> Son的实例
    // son.this -> new Father (); -> { age }
    super(44);
    // 放到上面不行 
    // { age hobby}
    this.hobby = "baskerball"; //Must call super constructor in derived class before accessing 'this' or returning from derived constructor
    console.log(this.age)
}
  // 访问的是son 原型上面的study 然后 访问到 Father 原型属性上面的swim
  study() {
    console.log(this, "this");
    // 沿着原型链找到Father，有就执行
    this.swim();
  }
}
const son = new Son();
son.study();

# call,apply,bind的区别

var obj = {
  a: 1,
};
var obj2 ={
    a:100
}
var a = 2;
function test(b,c) {
  // this默认-> 全局对象window
  console.log(this.a,b,c);
}
test();//2
test.call(obj);//1
test.apply(obj);//1
test.call(obj,3,4)//1 3 4
test.apply(obj,[3,4])//1 3 4

var test1 = test.bind(obj,3,4);
test1();
// bind 只会生效一次
var test2 = test1.bind(obj2,3,4) //1 3 4
test2();

var t = test.bind(obj2,3,4).bind(obj,3,4)//只会执行前面
console.log(t);

# this的基本准则

/**
 * this的基本原则： 谁调用this的宿主，this指向谁
 */

//  const test2 = ()=>{
//     console.log(this.b);
// }
var obj = {
    a:1,//成员属性
    b:2, 
    //成员方法
    test:function (){
            console.log(this.a) //obj
    },
    test2:test2,
    c:{
        d:4,
        test3:function (params) {
            // this -> obj.c
            console.log(this)
            console.log(this.d)
        }
    }
}

function test2(params) {
    console.log(this.b)
}


// obj.test();
obj.test2();//undefined
obj.c.test3();
// 对象方法内部的this指向最近作用域的引用

var obj2 = {
    a:1,
    b:2,
    test3:function (params) {
        // 这个函数是孤立的 最近的是window
        function t(params) {
            console.log(this)//window
        }
        t();
    }
}

obj2.test3();



obj2.__proto__={
    e:20
}

console.log(obj2.e);

var obj3 = Object.create({ //obj.proto挂一个和属性
    test4:function (params) {
        console.log(this.a+this.b)
    }
})
obj3.a=1;
obj3.b=2;
obj3.test4();


// var obj3 = {
//     test4:function (params) {
//         console.log(this.a+this.b)
//     }
// }

 * 1. test4 由obj3调用
 * 2. obj3就是test4最近的引用
 * 3. test4 this ->obj3
 * 4 obj3 = {  test 不存在 }

;(function (doc) {
    function Plus(a,b) {
        this.a=a;
        this.b=b;
    }
    Plus.prototype.init = function (params) {
        this.bindEvent();
    }
    Plus.prototype.bindEvent = function (){
        var _self = this;
        // obtn.addEventListener('click',this.handleCilck.bind(this),false)
        obtn.addEventListener('click',function (){
            _self.handleCilck();
        },false)
        // obtn.addEventListener('click',this.handleCilck,false)
    }
    Plus.prototype.handleCilck = function (){
        // this指向按钮 需要原型上的属性
        console.log(this.a+this.b)
    }
    window.Plus =Plus
})(document);

new Plus(3,4);

# 实例方法+继承

/** 
 * 类中是严格模式
 * */ 
/** 
 * 父亲有个吃水果的方法 有一个水果
 * 儿子有自己的水果 使用父亲吃水果的方法吃自己的水果
 * */ 

class Father {
    constructor(){
        //函数内部的this指向固定
        this.eat = this.eat.bind(this)//永远指向父亲
    }

    get Fruit (){
        return 'apple' 
    }
    eat (){
        console.log('I am eating a' +this.Fruit)
    }
}

class Son {
    get Fruit (){
        return 'pear'
    }
}
const father =  new Father();
const eat = new Son ();

son.eat = father.eat ()

son.eat();
// 儿子必须吃父亲的水果
