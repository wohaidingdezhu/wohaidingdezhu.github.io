深入浅出Promise对象

xijiajie 2021-12-13 JavaScript

回调当中嵌套回调，也称回调地狱。这种代码的可读性和可维护性都是非常差的，因为嵌套的层级太多。而且还有一个严重的问题，就是每次任务可能会失败，需要在回调里面对每个任务的失败情况进行处理，增加了代码的混乱程度。

# 功能点

/**
 * promise 承诺
 * 实现承诺      承诺石沉大海     承诺等待结果中
 * resovle        reject
 * 解决问题      拒绝考虑            pending 苦苦等待...
 * 解决异步流程化的一种手段    异步：各干个的
 * promise本身不是异步
 * 是一个构造函数 需要 new
 * promise 有且只有一个参数  excutor 执行器
 * excutor -> resolve reject 参数 都是函数
 * excutor 在new Promise 调用
 * let promise = new Promise ( ( resolve,reject )=>{   } )
 * excutor 是同步执行的 , then是异步调用
 * error只执行一个，具有冒泡性质,会一直穿透，then没有处理会走catch
 * */

# 用 Promise 完成 文件读写

// 用Promise 来解决文件读写

const { rejects } = require("assert");
const fs = require("fs");
const { resolve } = require("path");

let uid = 3;

function readFile(path, prevData) {
  return new Promise((resolve, reject) => {
    fs.readFile(path, "utf-8", function (err, data) {
      if (err) {
        reject(err);
      }
      const resData = JSON.parse(data);
      resolve({
        prevData,
        resData,
      });
    });
  });
}
readFile("./data/user.json")
  .then((res) => {
    const { resData } = res,
      userInfo = resData.filter((item) => item.id === uid)[0];
    // throw new Error ('失败了')
    return readFile("./data/userCourse.json", userInfo);
  })
  .then((res) => {
    const { prevData, resData } = res,
      userId = prevData.id;
    userCourse = resData.filter((item) => item.uid === userId)[0];
    return readFile("./data/course.json", {
      username: prevData.username,
      userCourse,
    });
  })
  .then((res) => {
    const { prevData, resData } = res,
      userCourses = prevData.userCourse.courses;
    let _arr = [];
    userCourses.map((id) => {
      resData.map((item) => {
        if (item.id === id) {
          _arr.push(item);
        }
      });
    });

    const userCourseInfo = {
      username: prevData.username,
      courses: _arr,
    };

    fs.writeFileSync(
      `./data/${userCourseInfo.username}.json`,
      JSON.stringify(userCourseInfo)
    );
  })
  .catch((err) => {
    console.log(err);
  });

# Promise.race()

const { log } = require("console");
const fs = require("fs");

function readFile(path, isSetError) {
  return new Promise((resolve, reject) => {
    fs.readFile(path, "utf-8", function (err, data) {
      if (err || isSetError) {
        reject("承诺石城大海");
      }

      const resData = JSON.parse(data);
      resolve(resData);
    });
  });
}

Promise.race([
  readFile("./data/user.json"),
  readFile("./data/userCourse.json"),
  readFile("./data/course.json", true),
])
  .then((res) => {
    console.log(res);
  })
  .catch((err) => {
    console.log(err);
  });

// 谁先完成就返回那个promise对象，无论是fullfilled还是rejected
// 如果是空，就什么都没返回
// 测试资源或者接口的响应速度

const URL = "";

const preloadImage = async (src) => {
  return new Promise((resolve) => {
    const img = new Image();
    img.src = src;
    if (img.complete) {
      resolve();
    }
    img.onload = () => {
      resolve();
    };
  });
};

function timeout() {
  return new Promise((reslove, reject) => {
    settimeout(() => {
      reject("图片请求失败了");
    }, 3000);
  });
}

function loadImage(src){
  var p=new Promise(function(resolve,reject){
      var img=new Image();
      // resolve 是表示正确完成时执行的函数
      img.onload=function(){
          resolve(img);
      }
      // reject 时表示错误时执行的函数
      img.onerror=function(){
          reject(src);
      }
      img.src=src;
  });
  return p;
}


Promise.race([await preloadImage(URL), timeout()])
  .then((res) => {
    console.log(res);
  })
  .catch((err) => {
    console.log(err);
  });
